##### metode #####
n = q*p
phi(n) = (p-1)(q-1)
0 < e < phi(n)
d := ed (mod phi(n)) = 1

offentlig: n e
hemmelig: d

kryptering
m -> c = m^e (mod n)

dektyptering
c -> m = c^d (mod n)


c^d (mod n) = (m^e (mod n))^d (mod n) = (m^e)^d (mod n) = m^ed (mod n) = m
